//	{
//		"type"				"announce" // "Type" of fake/force event to perform.

//		"override"			"faf_fake" // The override that donators need to use this command.

//		"cooldown"			"30" // After using this command, donators must wait this long, in seconds, before using FaF again. Admins can always use commands that they have override_admin access to.

//		// "ANNOUNCE" KEYS //
//		"items"				"1" // How many "item" arguments the command will take. Normally, this should be 1.

//								// However, if you want to just use 1, you can leave this key out.
//		"text"				"{PLAYER} has been chosen to demonstrate the example {ITEM1}!"
//									   // Text to print to the chat. You can use {PLAYER} (the target's name),
//									   // {ITEM1} (item names, e.g. /fakeexample sometarget Item Name),
//									   // any More Colours colour (e.g. {RED}) or a hexidecimal colour (e.g. {FF4040}),
//									   // {UNIQUE}, {VINTAGE}, {STRANGE}, {GENUINE}, {UNUSUAL}, {HAUNTED}, {VALVE}, {COMMUNITY}, or {COLLECTOR} for a quality colour.
//		"quality"			"1" // If 1, item names will be coloured yellow (Unique colour). If an item name begins
//								// with a quality word, e.g. "Strange ", "Unusual ", etc., it will be stripped from the item name,
//								// and the appropriate quality colour will be applied instead.
//		"case"				"1" // If 1, makes item names ALWAYS ALLCAPS. If 0, always lowercase. *Don't specify "case" to ignore capitalization.*
//		"noteam"			"1" // DON'T apply team-colouring to {PLAYER}'s name.
//		"particle"			"set_taunt_saharan_spy" // Particle effect to apply to the target.
//		"particle_life"		"10.0" // How long the particle effect should last/stay parented to the player.
//		"particle_offs"		"0.0 0.0 60.0" // Amount, in Hammer units, to offset the particle's position.
//		"emit"				"vo/spy_sf12_goodmagic04.wav" // Sound to play from the target.
//		"sound"				"vo/announcer_success.wav" // Sound to play to all players.
//		"playgamesound"		"Announcer.TrKillTargetRandom" // Same as "sound" but uses the playgamesound console command instead of using EmitSoundToAll. Follows sound scripts.
//
//
//		// "COMMAND" KEYS //
//		"cmd"				"{PARAMS}" // The command to force. ...Yeah. You can put stuff before or after {PARAMS} to customize it a bit.
//									  // Currently, the only other variable is {PLAYER} and {PLAYERUID}; to fake most SourceMod commands,
//									  // you'll want to target them using their user id, for example "sm_noclip #{PLAYERUID}"
//		"force"				"1" // Most commands should be faked (0, default), but for some, you'll need to enable force on the command (1).
//								// 0 uses FakeClientCommand() -- the server *pretends* that the client used the command. For commands like say, taunt, SourceMod commands, etc.
//								// 1 uses ClientCommand() and attempts to *force* it on the client. Most commands will be blocked and warn the client, but there are a few that work,
//								// such as retry, play, and playgamesound. Type "findflags server_can_execute" into your client console for a list.
//
